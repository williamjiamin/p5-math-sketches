RippaMathCore.js

Interactive math visualizations built on p5.js

Interactive Number Line

Drag the arrow marker along the line. Press SPACE to toggle snap-to-tick. Scroll (or pinch on mobile) to zoom. Sparkle animations play when you land on an integer.

let nl;
function setup() {
  createCanvas(800, 360);
  nl = new RippaMathCore.NumberLine({
    min: -10, max: 10,
    interactive: true,
    snap: false,
    zoom: true,
    animations: true,
    showArrow: true
  });
}
function draw()          { background(245); nl.draw(); }
function mousePressed()  { nl.mousePressed(); }
function mouseReleased() { nl.mouseReleased(); }
function keyPressed()    { nl.keyPressed(); }
function mouseWheel(e)   { return nl.mouseWheel(e); }

3D Coordinate System

Uses WEBGL. Drag (or swipe on mobile) to orbit the camera, scroll (or pinch) to zoom. Call c3.setPoint(x,y,z) to move the point.

let c3;
function setup() {
  createCanvas(700, 500, WEBGL);
  c3 = new RippaMathCore.Coordinate3D({
    gridMin: -5, gridMax: 5,
    freeOrbit: true
  });
  c3.setPoint(2, 3, 1);
}
function draw()          { background(240); c3.draw(); }
function mousePressed()  { c3.mousePressed(); }
function mouseDragged()  { c3.mouseDragged(); }
function mouseReleased() { c3.mouseReleased(); }
function mouseWheel(e)   { return c3.mouseWheel(e); }

Angle Explorer

Drag the purple handle to adjust the angle. See the angle name, degree measurement, and type. Tap quick-jump buttons for special angles. Toggle Snap for precise positioning.

let ae;
function setup() {
  createCanvas(900, 600);
  ae = new RippaMathCore.AngleExplorer({
    startAngle: 45,
    snapMode: false,
    showNaming: true,
    showTypeInfo: true,
    showQuickJump: true,
    pointLabels: ['A', 'B', 'C']
  });
}
function draw()          { background(248, 250, 255); ae.draw(); }
function mousePressed()  { ae.mousePressed(); }
function mouseDragged()  { ae.mouseDragged(); }
function mouseReleased() { ae.mouseReleased(); }
